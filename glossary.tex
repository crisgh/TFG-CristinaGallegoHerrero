\newglossaryentry{imdb}{%
  name={IMDb},
  description={Base de datos en línea que almacena información relacionada con películas, actores, series y programas de televisión, proporcionando resúmenes de tramas, clasificaciones y reseñas de fans y críticos de las mismas. Aunque inicialmente era un sitio operado por fans, en 1998 fue comprada por Amazon y actualmente su base de datos almacena información acerca de aproximadamente 6.5 millones de títulos y 10.4 millones de personalidades, además de contar con 83 millones de fans registrados.}
}

}

\newglossaryentry{mapreduce}{%
  name={MapReduce},
  description={Modelo de programación diseñado para dar soporte a la computación paralela sobre grandes conjuntos de datos distribuidos en un cluster. El nombre del framework se inspira en los nombres de dos importantes métodos de programación funcional: Map y Reduce. }
}
